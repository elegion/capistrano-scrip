task :setup_host do
  unless exists?(:deploy_user)
    set :deploy_user, user
    set :user, root_user
  end

  if remote_file_exists?(database_config_path)
    logger.important "Skipping creating DB config, file already exists: #{database_config_path}"
  else
    upload_config
    create_db_user
  end
end
